#!/bin/bash

BRIGHTNESS=$(echo -e "100\n75\n50\n25\n0" | rofi -dmenu -p "Select Brightness:")

if [[ "$BRIGHTNESS" =~ ^(100|75|50|25|0)$ ]]; then
  ddcutil --display 1 setvcp 10 $BRIGHTNESS
  ddcutil --display 2 setvcp 10 $BRIGHTNESS
  echo "Brightness of both monitors set to $BRIGHTNESS%"
else
  echo "Invalid brightness level"
  exit 1
fi
